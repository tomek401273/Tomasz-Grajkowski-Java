SELECT L.NAME, I.SUMMARY, I.DESCRIPTION 
FROM  ISSUESS I, ISSUESLIST L
WHERE L.ID = I.ISSUESLIST_ID;

SELECT U.FIRSTNAME, U.LASTNAME, I.SUMMARY, I.DESCRIPTION 
FROM ISSUESS I, USERS U
WHERE U.ID =I.USER_ID_ASSIGNEDTO;

SELECT U.FIRSTNAME,U.LASTNAME, COUNT(*) AS "ASSIGNED TASK"
FROM USERS U, ISSUESS I
WHERE U.ID = I.USER_ID_ASSIGNEDTO
group by I.USER_ID_ASSIGNEDTO;
